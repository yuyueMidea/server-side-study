### “强关联、可扩展”的主从表（Header–Line）设计思路

下面一套“强关联、可扩展”的主从表（Header–Line）设计思路与4个可直接落地的例子（含 PostgreSQL DDL）。每个例子都体现 1—N 关系，并适当穿插 N—N、层级、自引用与审计/状态流等常见需求。

设计总原则：
- 命名与主键：表名用单数；主键优先 uuid（或 bigserial），从表用外键引用主表。
- 通用字段：tenant_id（多租户可选）、created_at、updated_at、created_by、updated_by、is_deleted（软删可选）。
- 状态与历史：主表只存“当前状态”，配套 *_status_log 记录历史（时间、操作者、备注）。
- 金额/数量：用最小货币单位（分/厘）存整数；数量用 numeric(18,6)；加 CHECK > 0。
- 约束优先：NOT NULL、UNIQUE、CHECK、FOREIGN KEY … ON UPDATE/DELETE；常用组合唯一键（如主表ID+行号）。
- 索引：外键列、查询高频的筛选字段、时间列；必要时部分索引/联合索引。
- 可扩展字段：留 extras jsonb 存非结构化扩展，避免频繁改表。

---

**① 电商订单（orders ⇢ order_items）**

典型主从：订单头、订单行；外带支付、发货、状态流水、优惠券。
```
-- 参考：PostgreSQL
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 订单主表
CREATE TABLE orders (
  id           uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  tenant_id    uuid,
  order_no     text UNIQUE NOT NULL,               -- 业务单号
  buyer_id     uuid NOT NULL,
  currency     char(3) NOT NULL DEFAULT 'CNY',
  total_amount bigint NOT NULL CHECK (total_amount >= 0), -- 分
  status       text NOT NULL CHECK (status IN ('CREATED','PAID','SHIPPING','DONE','CANCELLED')),
  extras       jsonb NOT NULL DEFAULT '{}'::jsonb,
  created_at   timestamptz NOT NULL DEFAULT now(),
  updated_at   timestamptz NOT NULL DEFAULT now()
);

-- 订单行（从表）
CREATE TABLE order_items (
  id           uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  order_id     uuid NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  line_no      int  NOT NULL,
  sku_id       uuid NOT NULL,
  title        text NOT NULL,
  unit_price   bigint NOT NULL CHECK (unit_price >= 0),   -- 单价(分)
  qty          numeric(18,6) NOT NULL CHECK (qty > 0),
  line_amount  bigint NOT NULL CHECK (line_amount >= 0),
  extras       jsonb NOT NULL DEFAULT '{}'::jsonb,
  UNIQUE(order_id, line_no)
);

-- 支付记录（一个订单可多次支付/退款）
CREATE TABLE payments (
  id           uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  order_id     uuid NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  pay_no       text UNIQUE NOT NULL,
  channel      text NOT NULL CHECK (channel IN ('ALIPAY','WECHAT','STRIPE','PAYPAL','OFFLINE')),
  direction    text NOT NULL CHECK (direction IN ('PAY','REFUND')),
  amount       bigint NOT NULL CHECK (amount > 0),
  status       text NOT NULL CHECK (status IN ('INIT','SUCCESS','FAIL')),
  occurred_at  timestamptz NOT NULL DEFAULT now(),
  extras       jsonb NOT NULL DEFAULT '{}'::jsonb
);

-- 发货单（主从可拆，此处简化为一表）
CREATE TABLE shipments (
  id           uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  order_id     uuid NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  ship_no      text UNIQUE NOT NULL,
  carrier      text,
  tracking_no  text,
  status       text NOT NULL CHECK (status IN ('CREATED','OUT','DELIVERED')),
  shipped_at   timestamptz,
  delivered_at timestamptz,
  extras       jsonb NOT NULL DEFAULT '{}'::jsonb
);

-- 状态流
CREATE TABLE order_status_logs (
  id           uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  order_id     uuid NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  from_status  text,
  to_status    text NOT NULL,
  reason       text,
  changed_by   uuid,
  changed_at   timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX idx_order_items_order ON order_items(order_id);
CREATE INDEX idx_payments_order ON payments(order_id);
CREATE INDEX idx_shipments_order ON shipments(order_id);
CREATE INDEX idx_order_logs_order ON order_status_logs(order_id);
```
